vi dp;

int ans(int ind) {

	if (ind >= v.size()) return 0;
	// IF NOT -1 THEN I KNOW KI DP[IND] IS THE ANS ONLY
	if (dp[ind] != -1) return dp[ind];

	// MAKE TWO DECISION OF EITHER TAKE, OR DONT TAKE
	int take = 0, dontTake = 0;

	take = ans(ind + k + 1) + v[ind];

	//dont take
	dontTake = ans(ind + 1);
	return dp[ind] = max(take, dontTake);

}
